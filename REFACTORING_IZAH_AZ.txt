```typescript
export function QuestionPromptField({ question, onChange }) {
  switch (question.qtype) {
    case "ORDER_SENTENCE":
      return <PromptOrderSentence ... />;
    case "SHORT_TEXT":
      return <PromptShortText ... />;
    case "ESSAY":
      return <PromptEssay ... />;
    case "FILL_IN_BLANK":
      return <PromptFillInBlank ... />;
    case "DND_GAP":
      return <PromptDndGap ... />;
    case "INLINE_SELECT":
      return <PromptInlineSelect ... />;
    case "SPEAKING_RECORDING":
      return <PromptSpeakingRecording ... />;
    case "IMAGE_INTERACTIVE":
      return <PromptImageInteractive ... />;
    default:
      return <PromptDefault ... />;
  }
}
```
